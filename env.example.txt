# EmailJS Configuration
# Copy this file to .env.local and fill in your EmailJS credentials
# Get these values from https://www.emailjs.com/
# 
# After setting up your EmailJS account:
# 1. Go to Email Services and create a service (Gmail, Outlook, etc.)
# 2. Go to Email Templates and create a template
# 3. Go to Account > API Keys to get your Public Key
#
# Your EmailJS template should use these variables:
# - {{from_name}} - The sender's name
# - {{from_email}} - The sender's email
# - {{message}} - The message content
# - {{to_email}} - Your email address (contact@securebywinter.com)

NEXT_PUBLIC_EMAILJS_SERVICE_ID=your_service_id_here
NEXT_PUBLIC_EMAILJS_TEMPLATE_ID=your_template_id_here
NEXT_PUBLIC_EMAILJS_PUBLIC_KEY=your_public_key_here
